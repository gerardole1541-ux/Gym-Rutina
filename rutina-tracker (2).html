/** * BACKEND DE LA BITÁCORA DE GYM — Google Apps Script * --------------------------------------------------- * 1. Crea un Google Sheet nuevo (vacío, no necesitas armar columnas). * 2. Copia el ID del Sheet desde la URL: * https://docs.google.com/spreadsheets/d/ESTE_ES_EL_ID/edit * 3. En el Sheet: Extensiones > Apps Script. * 4. Borra el contenido de Code.gs y pega TODO este archivo. * 5. Reemplaza SHEET_ID abajo con el ID que copiaste. * 6. Deploy > New deployment > tipo "Web app". * - Execute as: Me * - Who has access: Anyone * 7. Copia la URL que te da (termina en /exec) y pégala en * CONFIG.SCRIPT_URL dentro de rutina-tracker.html */ const SHEET_ID = 'PEGA_AQUI_EL_ID_DE_TU_GOOGLE_SHEET'; const SESSIONS_SHEET = 'Sessions'; const SETS_SHEET = 'Sets'; function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureSheets_(ss); const action = (e.parameter.action || 'all'); if (action === 'all') { return jsonResponse_(getAllData_(ss)); } return jsonResponse_({ error: 'acción no reconocida: ' + action }); } function doPost(e) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureSheets_(ss); let body; try { body = JSON.parse(e.postData.contents); } catch (err) { return jsonResponse_({ error: 'JSON inválido' }); } const action = body.action; if (action === 'saveSession') { saveSession_(ss, body.payload); return jsonResponse_({ ok: true }); } if (action === 'deleteDay') { deleteDay_(ss, body.payload.day); return jsonResponse_({ ok: true }); } return jsonResponse_({ error: 'acción no reconocida: ' + action }); } // ---------- Helpers ---------- function ensureSheets_(ss) { if (!ss.getSheetByName(SESSIONS_SHEET)) { const sh = ss.insertSheet(SESSIONS_SHEET); sh.appendRow(['savedAt', 'day', 'date', 'location', 'rating', 'note', 'volume']); sh.setFrozenRows(1); } if (!ss.getSheetByName(SETS_SHEET)) { const sh = ss.insertSheet(SETS_SHEET); sh.appendRow(['savedAt', 'day', 'date', 'exerciseId', 'exerciseName', 'setNum', 'weight', 'reps', 'done']); sh.setFrozenRows(1); } } function saveSession_(ss, payload) { const sessSheet = ss.getSheetByName(SESSIONS_SHEET); const setsSheet = ss.getSheetByName(SETS_SHEET); const savedAt = new Date().toISOString(); sessSheet.appendRow([ savedAt, payload.day, payload.date, payload.location || 'gym', payload.rating, payload.note || '', payload.volume || 0, ]); const sets = payload.sets || []; const rows = sets.map(function (s, idx) { return [ savedAt, payload.day, payload.date, s.exId, s.exName || '', idx + 1, s.weight || '', s.reps || '', !!s.done, ]; }); if (rows.length > 0) { const startRow = setsSheet.getLastRow() + 1; setsSheet.getRange(startRow, 1, rows.length, rows[0].length).setValues(rows); } } function deleteDay_(ss, day) { [SESSIONS_SHEET, SETS_SHEET].forEach(function (name) { const sh = ss.getSheetByName(name); const data = sh.getDataRange().getValues(); // recorre de abajo hacia arriba para no romper índices al borrar for (let i = data.length - 1; i >= 1; i--) { if (data[i][1] === day) { sh.deleteRow(i + 1); } } }); } function getAllData_(ss) { const sessSheet = ss.getSheetByName(SESSIONS_SHEET); const setsSheet = ss.getSheetByName(SETS_SHEET); const sessValues = sessSheet.getDataRange().getValues(); const setValues = setsSheet.getDataRange().getValues(); const sessions = []; for (let i = 1; i < sessValues.length; i++) { const r = sessValues[i]; sessions.push({ savedAt: r[0], day: r[1], date: formatDate_(r[2]), location: r[3], rating: r[4], note: r[5], volume: r[6], }); } const sets = []; for (let i = 1; i < setValues.length; i++) { const r = setValues[i]; sets.push({ savedAt: r[0], day: r[1], date: formatDate_(r[2]), exId: r[3], exName: r[4], setNum: r[5], weight: r[6], reps: r[7], done: r[8], }); } return { sessions: sessions, sets: sets }; } function formatDate_(value) { // Google Sheets a veces convierte strings tipo "2026-07-15" en objetos Date. // Esto asegura que siempre regrese como string YYYY-MM-DD. if (value instanceof Date) { return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } return String(value); } function jsonResponse_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
